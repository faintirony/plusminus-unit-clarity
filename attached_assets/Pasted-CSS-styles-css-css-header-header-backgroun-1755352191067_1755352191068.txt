class ProductManager {
    constructor() {
        this.products = [...mockProducts];
        this.filteredProducts = [...this.products];
        this.currentFilters = {
            period: 'month',
            dateFrom: '2024-01-01',
            dateTo: '2024-12-31',
            search: '',
            marketplace: 'all',
            profitability: 'all'
        };

        this.init();
    }

    init() {
        this.setupEventListeners();
        this.updateStats();
        this.render();
    }

    setupEventListeners() {
        // –û–±—Ä–∞–±–æ—Ç—á–∏–∫ –ø–µ—Ä–∏–æ–¥–∞
        const periodSelect = document.getElementById('periodSelect');
        const dateFrom = document.getElementById('dateFrom');
        const dateTo = document.getElementById('dateTo');

        periodSelect.addEventListener('change', (e) => {
            this.handlePeriodChange(e.target.value);
        });

        dateFrom.addEventListener('change', (e) => {
            this.currentFilters.dateFrom = e.target.value;
            this.currentFilters.period = 'custom';
            periodSelect.value = 'custom';
            this.applyFilters();
        });

        dateTo.addEventListener('change', (e) => {
            this.currentFilters.dateTo = e.target.value;
            this.currentFilters.period = 'custom';
            periodSelect.value = 'custom';
            this.applyFilters();
        });

        // –ü–æ–∏—Å–∫
        const searchInput = document.getElementById('searchInput');
        searchInput.addEventListener('input', (e) => {
            this.currentFilters.search = e.target.value.toLowerCase();
            this.applyFilters();
        });

        // –§–∏–ª—å—Ç—Ä –º–∞—Ä–∫–µ—Ç–ø–ª–µ–π—Å–∞
        const marketplaceFilter = document.getElementById('marketplaceFilter');
        marketplaceFilter.addEventListener('change', (e) => {
            this.currentFilters.marketplace = e.target.value;
            this.applyFilters();
        });

        // –§–∏–ª—å—Ç—Ä –ø—Ä–∏–±—ã–ª—å–Ω–æ—Å—Ç–∏
        const profitabilityFilter = document.getElementById('profitabilityFilter');
        profitabilityFilter.addEventListener('change', (e) => {
            this.currentFilters.profitability = e.target.value;
            this.applyFilters();
        });

        // –≠–∫—Å–ø–æ—Ä—Ç
        const exportBtn = document.getElementById('exportBtn');
        exportBtn.addEventListener('click', () => {
            this.exportToExcel();
        });

        // –ú–æ–±–∏–ª—å–Ω–æ–µ –º–µ–Ω—é
        const mobileToggle = document.getElementById('mobileMenuToggle');
        const mobileFilters = document.getElementById('mobileFilters');

        mobileToggle.addEventListener('click', () => {
            mobileFilters.classList.toggle('open');
        });

        // –ó–∞–∫—Ä—ã—Ç–∏–µ –º–æ–±–∏–ª—å–Ω–æ–≥–æ –º–µ–Ω—é –ø—Ä–∏ –∫–ª–∏–∫–µ –≤–Ω–µ –µ–≥–æ

                mobileFilters.classList.remove('open');
            }
        });

    handlePeriodChange(period) {
        this.currentFilters.period = period;

        const today = new Date();
        const dateFrom = document.getElementById('dateFrom');
        const dateTo = document.getElementById('dateTo');

        switch(period) {
            case 'today':
                const todayStr = today.toISOString().split('T')[0];
                dateFrom.value = todayStr;
                dateTo.value = todayStr;
                this.currentFilters.dateFrom = todayStr;
                this.currentFilters.dateTo = todayStr;
                break;

            case 'yesterday':
                const yesterday = new Date(today);
                yesterday.setDate(yesterday.getDate() - 1);
                const yesterdayStr = yesterday.
